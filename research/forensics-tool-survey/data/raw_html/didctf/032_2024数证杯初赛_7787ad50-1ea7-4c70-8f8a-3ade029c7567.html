from hashlib import md5

def md5sum(s):
    _md5 = md5()
    _md5.update(s.encode('utf-8'))
    return _md5.hexdigest()

def bt_hash(password):
    md51 = md5sum(password)
    pwd_hash = md5sum(md5sum(md51+'_bt.cn'))
    return pwd_hash

print(bt_hash("jlb1998"))
